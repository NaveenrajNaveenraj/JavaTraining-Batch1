class PrimeNumber {

public static void main(String[] args)
{
int i=1,j=2;		
		for(i=1;i<=100;i++) 
		{
		 while(j<=9)
     	
  
	    if(i%j==0 || i%j==1)
	    {
	        System.out.println("prime number" +i);
	    }
     	
	    else 
	    {
	        System.out.println(" not prime number"+i);
     
     	}
	
	}

}
}
